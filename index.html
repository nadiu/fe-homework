git config --lobal user.name - встановлення імені користувача

git config --lobal user.email - встановлення адреси електронної пошти

git init - створює порожній репозиторій git

git status - для перегляду статусу файлів в яких були зміни

git add - додає всі файли

git commit -збереження зміни

git log - показує історію сомітів які були створені

git checkout - для переключення на існуючі вітку

git branch - для створення нової вітки

git push - для відправлення вітки на сервер

git pull - для отримання даних з серверу

git merge - для об'єднання декількох віток в одну

git clone - для створення копії існуючого git репозиторія

.gitignore - для приховування файлів і папок від git
